            document.getElementById('downloadModal').style.display = 'none';

        // Tab switching
        function showCodeTab(tab) {
            // Hide all tabs
            document.querySelectorAll('.code-content').forEach(content => {
                content.classList.remove('active');
            });
            
            document.querySelectorAll('.code-tab').forEach(tabEl => {
                tabEl.classList.remove('active');
            });
            
            // Show selected tab
            document.getElementById(tab + 'Code').classList.add('active');
            
            // Highlight tab
            event.currentTarget.classList.add('active');
        }
        
        // Load code files
        async function loadCodeFiles() {
            try {
                // Extract current HTML content
                const htmlContent = document.documentElement.outerHTML;
                document.getElementById('htmlCodeContent').textContent = htmlContent;
                
                // Fetch CSS content
                const cssResponse = await fetch('style.css');
                const cssContent = await cssResponse.text();
                document.getElementById('cssCodeContent').textContent = cssContent;
                
                // Fetch JS content
                const jsResponse = await fetch('main.js');
                const jsContent = await jsResponse.text();
                document.getElementById('jsCodeContent').textContent = jsContent;
            } catch (error) {
                console.error('Error loading code files:', error);
            }
        }
        
        // Download functions
        function downloadHTML() {
            const htmlContent = document.documentElement.outerHTML;
            downloadFile('index.html', htmlContent, 'text/html');
        }
        
        function downloadCSS() {
            fetch('style.css')
                .then(response => response.text())
                .then(cssContent => {
                    downloadFile('style.css', cssContent, 'text/css');
                })
                .catch(error => console.error('Error downloading CSS:', error));
        }
        
        function downloadJS() {
            fetch('main.js')
                .then(response => response.text())
                .then(jsContent => {
                    downloadFile('main.js', jsContent, 'text/javascript');
                })
                .catch(error => console.error('Error downloading JS:', error));
        }
        
        function downloadFile(filename, content, contentType) {
            const blob = new Blob([content], { type: contentType });
            const link = document.createElement('a');
            link.href = URL.createObjectURL(blob);
            link.download = filename;
            document.body.appendChild(link);
            link.click();
            document.body.removeChild(link);
        }
        
        async function downloadZIP() {
            try {

                    alert('ZIP libraries are not loaded properly. Please try again.');
                    return;
                }
                
                const zip = new JSZip();
                
                // Add HTML file
                const htmlContent = document.documentElement.outerHTML;
                zip.file('index.html', htmlContent);
                
                // Add CSS folder and file
                const cssResponse = await fetch('style.css');
                const cssContent = await cssResponse.text();
                zip.folder('css').file('style.css', cssContent);
                
                // Add JS folder and file
                const jsResponse = await fetch('main.js');
                const jsContent = await jsResponse.text();
                zip.folder('js').file('main.js', jsContent);
                
                // Generate the ZIP file
                zip.generateAsync({ type: 'blob' })
                    .then(function(content) {
                        // Use FileSaver.js to save the ZIP
                        saveAs(content, 'portfolio-source-code.zip');
                    });
            } catch (error) {
                console.error('Error creating ZIP file:', error);
                alert('Error creating ZIP file: ' + error.message);
            }
        }

// Initialize scroll animations
function initScrollAnimations() {
    const sections = document.querySelectorAll('section');
    
    function checkSections() {
        const triggerBottom = window.innerHeight * 0.85;
        
        sections.forEach(section => {
            const sectionTop = section.getBoundingClientRect().top;
            
            if (sectionTop < triggerBottom) {
                section.classList.add('fade-in');
                
                // Animate elements inside the visible section
                const elements = section.querySelectorAll('.animate');
                elements.forEach((element, index) => {
                    setTimeout(() => {
                        element.classList.add('visible');
                    }, index * 100);
                });
            }
        });
    }
    
    // Initial check
    checkSections();
    
    // Check on scroll
    window.addEventListener('scroll', checkSections);
}

// Initialize sticky header
function initStickyHeader() {
    const header = document.querySelector('header');
    const heroHeight = document.querySelector('.hero').offsetHeight;
    
    function checkHeader() {
        if (window.scrollY > 100) {
            header.classList.add('sticky');
        } else {
            header.classList.remove('sticky');
        }
    }
    
    // Initial check
    checkHeader();
    
    // Check on scroll
    window.addEventListener('scroll', checkHeader);
}

// Mobile menu toggle
function initMobileMenu() {
    const menuBtn = document.querySelector('.menu-btn');
    const closeBtn = document.querySelector('.nav-close');
    const navLinks = document.querySelector('.nav-links');
    const navItems = document.querySelectorAll('.nav-links a');
    
    if (menuBtn) {
        menuBtn.addEventListener('click', () => {
            navLinks.classList.add('show');
        });
    }
    
    if (closeBtn) {
        closeBtn.addEventListener('click', () => {
            navLinks.classList.remove('show');
        });
    }
    
    navItems.forEach(item => {
        item.addEventListener('click', () => {
            navLinks.classList.remove('show');
        });
    });
}

// Typing effect for hero section
class TypeWriter {
    constructor(txtElement, words, wait = 3000) {
        this.txtElement = txtElement;
        this.words = words;
        this.txt = '';
        this.wordIndex = 0;
        this.wait = parseInt(wait, 10);
        this.type();
        this.isDeleting = false;
    }
    
    type() {
        // Current index of word
        const current = this.wordIndex % this.words.length;
        // Get full text of current word
        const fullTxt = this.words[current];
        
        // Check if deleting
        if(this.isDeleting) {
            // Remove char
            this.txt = fullTxt.substring(0, this.txt.length - 1);
        } else {
            // Add char
            this.txt = fullTxt.substring(0, this.txt.length + 1);
        }
        
        // Insert txt into element
        this.txtElement.innerHTML = `<span class="txt">${this.txt}</span>`;
        
        // Initial Type Speed
        let typeSpeed = 300;
        
        if(this.isDeleting) {
            typeSpeed /= 2;
        }
        
        // If word is complete
        if(!this.isDeleting && this.txt === fullTxt) {
            // Make pause at end
            typeSpeed = this.wait;
            // Set delete to true
            this.isDeleting = true;
        } else if(this.isDeleting && this.txt === '') {
            this.isDeleting = false;
            // Move to next word
            this.wordIndex++;
            // Pause before start typing
            typeSpeed = 500;
        }
        
        setTimeout(() => this.type(), typeSpeed);
    }
}

// Init TypeWriter On DOM Load
document.addEventListener('DOMContentLoaded', function() {
    const txtElement = document.querySelector('.txt-type');
    if (txtElement) {
        const words = JSON.parse(txtElement.getAttribute('data-words'));
        const wait = txtElement.getAttribute('data-wait');
        // Init TypeWriter
        new TypeWriter(txtElement, words, wait);
    }
});

// Smooth scrolling for anchor links
document.querySelectorAll('a[href^="#"]').forEach(anchor => {
    anchor.addEventListener('click', function (e) {
        e.preventDefault();
        
        const targetId = this.getAttribute('href');
        if (targetId === '#') return;
        
        const targetElement = document.querySelector(targetId);
        if (targetElement) {
            window.scrollTo({
                top: targetElement.offsetTop - 70,
                behavior: 'smooth'
            });
        }
    });
});
